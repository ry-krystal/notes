      for(let i = 0; i < oldChildren.length; i++) {
        const oldVNode = oldChildren[i]
        // 拿旧子节点oldVNode 去新的一组子节点中寻找具有相同key值的节点
         const has = newChildren.find(vnode => vnode.key === oldVNode.key)

         if (!has) {
          // 如果没有找到具有相同key值的节点，则说明需要删除该节点
          // 调用unmount函数将其卸载
          unmount(oldVNode)
         }
      }
